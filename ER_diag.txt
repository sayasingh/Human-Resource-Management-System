Table users {
  id int [pk]
  username varchar
  password varchar
  role enum
}

Table employees {
  id int [pk]
  name varchar
  email varchar
  phone varchar
  department varchar
  designation varchar
  user_id int [ref: > users.id]
}

Table leave_requests {
  id int [pk]
  employee_id int [ref: > employees.id]
  reason text
  from_date date
  to_date date
  status enum
  created_at timestamp
}

Table attendance {
  id int [pk]
  employee_id int [ref: > employees.id]
  date date
  status enum
  created_at timestamp
}
